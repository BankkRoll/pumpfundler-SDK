This guide will walk you through the process of installing Pumpfundler SDK and setting up advanced examples in both Node.js and React environments.

## Prerequisites

Before you begin, make sure you have the following installed:

<Check>Node.js (version 14 or later)</Check>
<Check>npm (version 6 or later)</Check>

## Installation Steps

<Steps>
  <Step title="Create a new project directory">
    ```bash
    mkdir pumpfundler-project
    cd pumpfundler-project
    ```
  </Step>
  <Step title="Initialize a new Node.js project">
    ```bash
    npm init -y
    ```
  </Step>
  <Step title="Install Pumpfundler SDK and dependencies">
    ```bash
    npm install pumpfundler-sdk @solana/web3.js @coral-xyz/anchor
    ```
  </Step>
  <Step title="Configure your environment">
    Set up your Solana connection and wallet using a local private key:
    ```typescript
    import { Connection, Keypair } from "@solana/web3.js";
    import { AnchorProvider, Wallet } from "@coral-xyz/anchor";

    // Use your actual private key here
    const privateKey = "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef";
    const keypair = Keypair.fromSecretKey(Buffer.from(privateKey, 'hex'));

    const connection = new Connection("https://api.mainnet-beta.solana.com");
    const wallet = new Wallet(keypair);
    ```
  </Step>
  <Step title="Initialize the SDK">
    Create a PumpFundlerSDK instance:
    ```typescript
    import { PumpFundlerSDK, PumpFundlerConfig } from "pumpfundler-sdk";

    // Use your actual Jito auth private key here
    const jitoAuthPrivateKey = "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef";

    const config: PumpFundlerConfig = {
      connection,
      jitoFee: 1000000, // 0.001 SOL
      commitmentLevel: "confirmed",
      blockEngineUrl: "https://your-jito-block-engine-url.com",
      jitoAuthKeypair: Keypair.fromSecretKey(Buffer.from(jitoAuthPrivateKey, 'hex')),
    };

    const provider = new AnchorProvider(connection, wallet, {});
    const sdk = new PumpFundlerSDK(provider, config);
    ```
  </Step>
  <Step title="Create and Buy with Multiple Wallets">
    Use createAndBuy with multiple local wallets:
    ```typescript
    import { Keypair, PublicKey } from "@solana/web3.js";
    import { Wallet } from "@coral-xyz/anchor";
    import { CreateTokenMetadata } from "pumpfundler-sdk";

    // Use your actual private keys here
    const localPrivateKeys = [
      "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef",
      "2345678901abcdef2345678901abcdef2345678901abcdef2345678901abcdef",
      "3456789012abcdef3456789012abcdef3456789012abcdef3456789012abcdef",
      "4567890123abcdef4567890123abcdef4567890123abcdef4567890123abcdef",
      "5678901234abcdef5678901234abcdef5678901234abcdef5678901234abcdef"
    ];

    const localWallets = localPrivateKeys.map(privateKey => 
      new Wallet(Keypair.fromSecretKey(Buffer.from(privateKey, 'hex')))
    );

    const metadata: CreateTokenMetadata = {
      name: "My Token",
      symbol: "MTK",
      description: "A sample token",
      file: new Blob(["dummy image data"]),
    };

    async function createAndBuyWithLocalWallets() {
      const creator = localWallets[0];
      const mint = Keypair.generate();
      const buyers = localWallets.slice(1);
      const buyAmountSol = BigInt(1_000_000_000); // 1 SOL per buyer

      const result = await sdk.createAndBuy(
        creator.payer,
        mint,
        buyers.map(wallet => wallet.payer),
        metadata,
        buyAmountSol * BigInt(buyers.length),
        300n // 3% slippage
      );

      console.log("Create and buy result with local wallets:", result);
    }

    createAndBuyWithLocalWallets().catch(console.error);
    ```
  </Step>
</Steps>

## Full Node.js Example

Here's a complete Node.js example that demonstrates how to use the Pumpfundler SDK:

```typescript
import { Connection, Keypair, PublicKey } from "@solana/web3.js";
import { AnchorProvider, Wallet } from "@coral-xyz/anchor";
import { PumpFundlerSDK, PumpFundlerConfig, CreateTokenMetadata } from "pumpfundler-sdk";

// Note: Using hardcoded private keys is not secure for production use
const CREATOR_PRIVATE_KEY = "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef";
const BUYER_PRIVATE_KEYS = [
  "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef",
  "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef",
  "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef",
  "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef",
  "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef"
];
const JITO_AUTH_PRIVATE_KEY = "1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef";

async function main() {
  const connection = new Connection("https://api.mainnet-beta.solana.com");

  const creator = Keypair.fromSecretKey(Buffer.from(CREATOR_PRIVATE_KEY, 'base64'));
  const buyers = BUYER_PRIVATE_KEYS.map(key => Keypair.fromSecretKey(Buffer.from(key, 'base64')));
  const jitoAuthKeypair = Keypair.fromSecretKey(Buffer.from(JITO_AUTH_PRIVATE_KEY, 'base64'));

  const config: PumpFundlerConfig = {
    connection,
    jitoFee: 1_000_000, // 0.001 SOL
    commitmentLevel: "confirmed",
    blockEngineUrl: "https://your-jito-block-engine-url.com",
    jitoAuthKeypair: jitoAuthKeypair.secretKey.toString('base64'),
  };

  const wallet = new Wallet(creator);
  const provider = new AnchorProvider(connection, wallet, {});
  const sdk = new PumpFundlerSDK(provider, config);

  const mint = Keypair.generate();
  
  const metadata: CreateTokenMetadata = {
    name: "Example Token",
    symbol: "EXT",
    description: "An example token created with Pumpfundler SDK",
    file: new Blob(["Example token image data"]),
    twitter: "https://twitter.com/exampletoken",
    telegram: "https://t.me/exampletoken",
    website: "https://www.exampletoken.com",
  };

  const buyAmountSol = BigInt(5_000_000_000); // 5 SOL total
  const slippageBasisPoints = 300n; // 3% slippage tolerance

  try {
    const result = await sdk.createAndBuy(
      creator,
      mint,
      buyers,
      metadata,
      buyAmountSol,
      slippageBasisPoints
    );

    console.log("Token creation and initial buy result:", result);

    // Execute additional buy orders
    for (const buyer of buyers) {
      const buyResult = await sdk.buy(
        buyer,
        mint.publicKey,
        BigInt(1_000_000_000), // 1 SOL per buyer
        200n // 2% slippage for individual buys
      );
      console.log(`Additional buy result for \${buyer.publicKey.toBase58()}:`, buyResult);
    }

    // Set up event listeners
    sdk.addEventListener("createEvent", (event, slot, signature) => {
      console.log("New token created:", event.name, "at slot", slot, "with signature", signature);
    });

    sdk.addEventListener("tradeEvent", (event, slot, signature) => {
      console.log(
        `Trade event: \${event.isBuy ? "Buy" : "Sell"} of \${event.tokenAmount} tokens`,
        `for \${event.solAmount} SOL at slot \${slot}`
      );
    });

  } catch (error) {
    console.error("Error in create and buy process:", error);
  }
}

main().catch(console.error);
```

<Warning>
  Remember that this example uses hardcoded private keys for demonstration purposes. In a real application, you should never expose private keys in your frontend code. Instead, use secure key management solutions and handle sensitive operations on a secure backend.
</Warning>
